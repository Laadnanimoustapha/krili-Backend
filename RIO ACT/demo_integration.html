üîó Integration with KRILI Chat

To integrate this moderation system with your KRILI chat, add this code to your Chat.html:

<script> // KRILI Chat Moderation Integration class KRILIModerator { constructor() { this.apiUrl = 'http://localhost:5000'; this.setupModeration(); } async moderateMessage(message, userId) { const response = await fetch(`${this.apiUrl}/moderate`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ message, user_id: userId }) }); return await response.json(); } async setupModeration() { const sendBtn = document.querySelector('.send-btn'); const messageInput = document.querySelector('.message-input'); if (sendBtn && messageInput) { sendBtn.onclick = async (e) => { e.preventDefault(); const message = messageInput.value.trim(); if (!message) return; const result = await this.moderateMessage(message, 'user123'); if (result.is_flagged && result.action === 'block') { alert(result.warning_message); messageInput.value = ''; return; } // Send message normally this.sendMessage(message); messageInput.value = ''; }; } } sendMessage(message) { // Your existing message sending logic here console.log('Sending message:', message); } } // Initialize when page loads new KRILIModerator(); </script>
